一、重新理解 React 的心智模型

1.1 React 不是响应式，是快照（State → Snapshot → UI）

// React 的核心循环：trigger → render → commit

function Counter() {
  const [count, setCount] = useState(0);

  // ❌ React 新手常犯的错误：以为 setState 后立即能读新值
  function handleClick() {
    setCount(count + 1);
    console.log(count);  // 还是旧值！渲染还没发生
  }

  // ✅ 正确理解：count 是本次渲染的快照，不会在渲染中途突变
  // 下一次渲染时，React 用新的 props/state 再跑一遍函数 → 得到新快照

  return <button onClick={handleClick}>{count}</button>;
}

1.2 闭包陷阱——Hooks 时代最大的心智负担

function Timer() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    const id = setInterval(() => {
      // ❌ 问题：这个 effect 只在 mount 时运行一次
      //    闭包捕获的 count 永远是 0
      setCount(count + 1);  // 永远是 setCount(0 + 1) = 1
    }, 1000);
    return () => clearInterval(id);
  }, []); // ← 空依赖

  // ✅ 解法一：函数式更新（推荐）
  setCount(c => c + 1);  // c 总是最新值，不依赖闭包

  // ✅ 解法二：正确声明依赖（但会导致 interval 反复创建/清理）
  // 每次 count 变 → effect 重新运行 → clearInterval → setInterval
  // 一般不推荐用于 interval
}, [count]);

  // ✅ 解法三：用 useRef 绕开闭包
  const countRef = useRef(count);
  countRef.current = count;  // 每次渲染更新 ref
  useEffect(() => {
    const id = setInterval(() => setCount(countRef.current + 1), 1000);
    return () => clearInterval(id);
  }, []);
⚠️ 核心原理：每次渲染都是一次独立的函数调用，Hooks 的值是那次调用时的快照。useEffect / useCallback / useMemo 的闭包里捕获的是创建时的值，不是最新值。这就是"过期闭包"问题的根源。

二、Hooks 深度解析（每个都要懂原理）

2.1 useState —— 挂载时与更新时的不同行为

// mini-react 中 useState 的简化实现
let currentFiber;          // 当前正在处理的 Fiber 节点
let hookIndex = 0;         // 当前 hook 在链表中的位置

function useState(initial) {
  const fiber = currentFiber;
  const hook = fiber.memoizedState
    ? fiber.memoizedState[hookIndex]
    : null;

  // === 挂载阶段 ===
  if (!hook) {
    const state = typeof initial === 'function' ? initial() : initial;
    fiber.memoizedState[hookIndex] = {
      state,
      queue: [],        // 更新队列（setState 的调用）
      next: null,
    };
  }

  const hookState = fiber.memoizedState[hookIndex];
  const currentIdx = hookIndex;  // 闭包捕获当前位置

  // 派发更新
  const dispatch = (action) => {
    const hook = fiber.memoizedState[currentIdx];
    // 加入更新队列
    hook.queue.push(
      typeof action === 'function' ? action : () => action
    );
    // 调度重渲染
    scheduleUpdate(fiber);
  };

  // === 更新阶段：依次执行队列中的更新 ===
  if (fiber.isUpdating) {
    let newState = hookState.state;
    hookState.queue.forEach(updateFn => {
      newState = updateFn(newState);  // 依次应用，每个拿到的是上一个的结果
    });
    hookState.state = newState;
    hookState.queue = [];  // 清空队列
  }

  hookIndex++;
  return [hookState.state, dispatch];
}

2.2 useEffect vs useLayoutEffect：微妙的时序差

// 完整渲染流程中的时序：

render 阶段（可中断，可恢复，React 内部）
  ↓ 计算新 Fiber 树，diff 旧 Fiber 树

commit 阶段（不可中断，必须原子完成）
  ├─ Before Mutation：getSnapshotBeforeUpdate (class)
  ├─ Mutation：DOM 操作（增删改属性）
  │    ├─ useLayoutEffect 的 cleanup  ← 同步执行！
  │    └─ useLayoutEffect 的 setup    ← 同步执行，阻塞浏览器绘制！
  └─ Layout：布局计算完成

浏览器绘制（paint）← 用户看到新 UI

useEffect 的 cleanup + setup  ← 异步执行，不阻塞绘制！

// 何时用哪个？
// useLayoutEffect：需要同步读取/修改 DOM 布局 → 如测量元素尺寸、滚动位置
// useEffect：数据请求、订阅、日志、副作用 → 99% 的情况用这个

2.3 useMemo vs useCallback —— 别说"只是语法糖"

useMemouseCallback
缓存什么值（任何类型）函数引用
本质useMemo(fn, deps) ≈ useState(() => fn())[0]（但每次渲染重新求值）useCallback(fn, deps) ≈ useMemo(() => fn, deps)
真正用途避免昂贵的重复计算保持函数引用稳定（避免子组件不必要的重渲染 + 作为 effect 依赖）
不要滥用简单计算 memo 的开销 > 重算的开销props 直接传函数也可以，只在子组件用了 React.memo 时才有效
// ❌ 滥用 useCallback：子组件没 memo，缓存等于白做
const handleClick = useCallback(() => { doSomething(id); }, [id]);
return <Child onClick={handleClick} />;  // Child 没 memo → 还是会重渲染

// ✅ 正确用法：配合 React.memo
const MemoChild = React.memo(Child);  // memo 会比较 props 是否变化
const handleClick = useCallback(() => { doSomething(id); }, [id]);
return <MemoChild onClick={handleClick} />;  // handleClick 稳定 → Child 不重渲染

// ⭐ 更推荐的模式：把状态下沉（state down）
// 不把 handleClick 传给深层组件，而是让深层组件自己管理自己的状态

2.4 useRef —— 不只是"拿到 DOM 元素"

// useRef 的三个核心用途：

// 1. DOM 引用（最常见）
const inputRef = useRef<HTMLInputElement>(null);
useEffect(() => { inputRef.current?.focus(); }, []);

// 2. 跨渲染保持可变值（不触发重渲染）
const prevCountRef = useRef(count);
useEffect(() => { prevCountRef.current = count; });  // 更新后保存

// 3. 保存最新的回调引用（避免过期闭包）
const latestCallback = useRef(onChange);
latestCallback.current = onChange;  // 每次渲染更新
// 在 effect 或事件处理器里始终调 latestCallback.current → 拿到的总是最新函数

2.5 useReducer —— 复杂状态逻辑的正确选择

// 不是 Redux 的替代品，而是 useState 的升级版
function reducer(state, action) {
  switch (action.type) {
    case 'increment': return { ...state, count: state.count + 1 };
    case 'decrement': return { ...state, count: state.count - 1 };
    case 'reset':    return { count: action.payload };
  }
}

// 关键优势：dispatch 的引用是稳定的（不像 setState 来自闭包）
const [state, dispatch] = useReducer(reducer, { count: 0 });
// dispatch 永远指向同一个函数 → 可以作为 effect 的依赖而不会导致重复执行

三、Fiber 架构 —— React 的骨架

3.1 为什么需要 Fiber？

// 旧架构（Stack Reconciler）：递归遍历，不可中断
// 一棵 10 万节点的树更新 → JS 线程阻塞数百毫秒 → 页面卡顿

// Fiber 架构：可中断、可分片的工作单元
// 每个 Fiber 节点是一个小任务，React 可以：
//   1. 暂停当前工作（让出主线程给用户交互）
//   2. 继续之前的工作
//   3. 丢弃已完成的工作（如果有更高优先级的更新）

3.2 Fiber 节点的数据结构

type Fiber = {
  // === 静态结构（对应组件树） ===
  tag: WorkTag,              // 类型：FunctionComponent / ClassComponent / HostComponent...
  key: null | string,
  type: any,                 // 对于 HostComponent 是标签名，对于 FC 是函数本身
  stateNode: any,            // 对应的 DOM 节点或类组件实例

  return: Fiber | null,      // ← 父 Fiber（单向链表，return 指向父）
  child: Fiber | null,       // → 第一个子 Fiber
  sibling: Fiber | null,     // → 下一个兄弟 Fiber

  // === 动态工作状态 ===
  pendingProps: any,         // 新的 props
  memoizedProps: any,        // 已生效的 props
  memoizedState: any,        // 已生效的 state（hooks 链表挂在这）
  updateQueue: any,          // 待处理的更新

  // === 副作用 ===
  flags: Flags,              // 副作用标记（Placement / Update / Deletion...）
  subtreeFlags: Flags,       // 子树中所有副作用标记的汇总
  deletions: Fiber[] | null, // 待删除的子节点

  // === 调度优先级 ===
  lanes: Lanes,              // 本节点的优先级
  childLanes: Lanes,         // 子树中所有优先级的汇总

  // === 双缓冲 ===
  alternate: Fiber | null,   // 指向另一棵树中的对应节点（current ↔ workInProgress）
};

3.3 双缓冲（Double Buffering）—— current 树 vs workInProgress 树

// React 同时维护两棵 Fiber 树：
//
//  current tree       → 屏幕上当前显示的内容（已经提交的）
//  workInProgress tree → 正在内存中构建的新版本
//
// 每个 Fiber 节点都有 .alternate 指向另一棵树中对应的节点
// commit 完成后，两棵树角色互换（current ↔ workInProgress）

function createWorkInProgress(current, pendingProps) {
  let workInProgress = current.alternate;
  if (!workInProgress) {
    // 首次渲染：创建新的
    workInProgress = createFiber(current.tag, pendingProps, current.key);
    workInProgress.alternate = current;
    current.alternate = workInProgress;
  } else {
    // 更新：复用了已有的 alternate，重置部分字段
    workInProgress.pendingProps = pendingProps;
    workInProgress.flags = NoFlags;
    workInProgress.subtreeFlags = NoFlags;
    workInProgress.deletions = null;
  }

  // 复用上次的状态（state / hooks 链表等）
  workInProgress.memoizedState = current.memoizedState;
  return workInProgress;
}
为什么用双缓冲？在构建新树的过程中，旧树仍然完整可用（屏幕上的 UI 不受影响）。构建完成后一次性 commit（切换指针），整个过程原子且无闪烁。

3.4 workLoop —— 可中断的渲染循环

function workLoop(deadline) {
  let shouldYield = false;

  // 深度优先遍历 Fiber 树，逐个处理每个节点
  while (nextUnitOfWork && !shouldYield) {
    nextUnitOfWork = performUnitOfWork(nextUnitOfWork);
    shouldYield = deadline.timeRemaining() < 1;  // 时间片用完了，让出主线程
  }

  if (!nextUnitOfWork && workInProgressRoot) {
    // 整棵树处理完毕 → 进入 commit 阶段
    commitRoot();
  }

  // 如果还有工作没做完且需要让出 → 通过 scheduler 调度下一次继续
  if (nextUnitOfWork) {
    requestIdleCallback(workLoop);
  }
}

function performUnitOfWork(fiber) {
  // 1. beginWork：处理当前节点（diff props、调用组件函数、生成子节点）
  beginWork(fiber);

  // 2. 深度优先：有子节点 → 处理子节点
  if (fiber.child) return fiber.child;

  // 3. 没有子节点 → 回溯，完成节点（completeWork：创建 DOM、收集副作用）
  let next = fiber;
  while (next) {
    completeWork(next);
    if (next.sibling) return next.sibling;  // 有兄弟 → 处理兄弟
    next = next.return;  // 没有兄弟 → 回到父节点
  }
}

四、Reconciliation（Diff）—— 精确理解

4.1 Diff 的三条假设

// React 的 diff 基于三个启发式假设（降低 O(n³) → O(n)）：

// 假设 1：不同类型 → 整棵树重建
<div>  <span />  </div>          // 旧
<header>  <span />  </header>      // 新 → div → header，类型变了 → 整个子树卸载重建

// 假设 2：通过 key 标识"同一个"元素
// 没有 key → 按索引比较 → 列表重排会大量错误复用
// 有 key   → 按 key 匹配 → 精确知道哪个移动了、哪个新增了/删除了

// 假设 3：同类型元素 → 只更新变化的属性
<div className="old" title="old" />
<div className="new" title="old" />  → 只更新 className

4.2 key 的正确使用

// ❌ 用 index 作 key（列表排序/增删时会错乱）
{list.map((item, index) => <Item key={index} {...item} />)}

// 问题：删除第 1 项 → index 全变了 → React 认为每一项都变了
// → 可能触发不必要的不稳定（input 焦点丢失、动画错乱、state 错误复用）

// ✅ 用稳定唯一 ID 作 key
{list.map(item => <Item key={item.id} {...item} />)}

// ⭐ key 只在同一层兄弟节点间有意义，不是全局的

5.1 选择状态管理方案的决策树

你的状态是哪种？
├─ 服务端缓存（API 响应）             → React Query / SWR

六、新的范式：React Server Components

6.1 RSC 的核心思想：组件不在客户端运行

// Server Component（无交互，在服务端渲染）
async function ProductList() {
  const products = await db.query('SELECT * FROM products');
  //        ↑ 直接在组件里查数据库！没有 API 层
  return (
    <ul>
      {products.map(p => (
        <li key={p.id}>
          {p.name}
          <AddToCartButton productId={p.id} />  {/* Client Component */}
        </li>
      ))}
    </ul>
  );
}

// Client Component（有交互，在浏览器运行）
'use client';
function AddToCartButton({ productId }) {
  const [loading, setLoading] = useState(false);
  return (
    <button onClick={async () => {
      setLoading(true);
      await addToCart(productId);
      setLoading(false);
    }}>
      {loading ? 'Adding...' : 'Add to Cart'}
    </button>
  );
}

6.2 RSC vs SSR vs CSR —— 别再搞混

CSR (客户端渲染)SSR (服务端渲染)RSC (React Server Components)
HTML 生成空壳 HTML，JS 加载后渲染服务端生成完整 HTML服务端生成组件树（不是 HTML）
水合不需要需要 Hydration不需要（Server Component 不水合）
JS 体积大（全部组件代码）大（全部组件代码）小（Server Component 代码不发送）
直接访问后端❌ 需要 API❌ 需要 API（getServerSideProps）✅ 可直接查 DB、读文件
交互✅✅ 水合后❌（需配合 Client Component）
⭐ RSC 改变了一切：以前 React 必须把所有组件代码打包发给浏览器。现在只有 Client Component（带交互的）的代码需要发给浏览器，Server Component 的逻辑完全在服务端执行，不增加客户端 JS 体积。这是 React 生态最根本的范式转移。

7.2 优化手段速查表

手段原理何时用代价
React.memo Shallow props 对比，不变就跳过 子组件纯展示、props 变化少 每次渲染做一次比较的开销
useMemo 缓存计算结果 计算开销大（>1ms）的派生值 内存 + 依赖比较开销
useCallback 稳定函数引用 函数作为 memo 子组件的 prop 同上
React.lazy 组件懒加载 路由级拆分、非首屏组件 首次加载时有网络 + 解析延迟
startTransition 标记为非紧急更新 搜索输入框（输入不卡） 需要 React 18+
useDeferredValue 延迟更新某值 列表过滤、搜索建议 可能短暂显示旧值
状态下沉（state down） 把状态放在需要的最小范围 表单值不必上升到全局 无运行时开销（最佳）
内容提升（lift content） 把不依赖状态的 JSX 作为 children 传入 弹窗包裹的列表内容不因为弹窗开关而重渲染 无运行时开销（最佳）

7.3 状态下沉 vs 内容提升——无开销的优化

// ❌ 写法：弹窗的状态变化导致 ExpensiveTree 也重渲染
function App() {
  const [open, setOpen] = useState(false);
  return (
    <div>
      <button onClick={() => setOpen(true)}>Open</button>
      {open && <Modal />}
      <ExpensiveTree />   <!-- App 重渲染 → ExpensiveTree 重渲染 -->
    </div>
  );
}

// ✅ 写法二选一：
// 方案 A：状态下沉 — open 下放到真正需要的组件
function App() {
  return (
    <div>
      <ModalTrigger />      <!-- open 状态在这内部 -->
      <ExpensiveTree />     <!-- App 不会因 open 变化而渲染 -->
    </div>
  );
}

// 方案 B：内容提升 — ExpensiveTree 作为 children 传入
function ModalWrapper({ children }) {
  const [open, setOpen] = useState(false);
  return (
    <>
      <button onClick={() => setOpen(true)}>Open</button>
      {open && <Modal />}
      {children}      <!-- children 在父组件中创建，不受 ModalWrapper 重渲染影响 -->
    </>
  );
}

function App() {
  return (
    <ModalWrapper>
      <ExpensiveTree />   <!-- 在 App 中创建，不会因为 ModalWrapper 的 open 变化而重建 -->
    </ModalWrapper>
  );
}

// useDeferredValue：延迟某个值的更新
function List({ query }) {
