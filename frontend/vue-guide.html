一、响应式系统（Vue 的灵魂）

1.1 Vue 3 响应式原理（必须能白板画出）

// 核心三步：track → trigger → effect

// 1. 依赖收集 (track)
let activeEffect;
const targetMap = new WeakMap();  // target → depsMap

function track(target, key) {
  let depsMap = targetMap.get(target);
  if (!depsMap) { targetMap.set(target, (depsMap = new Map())); }
  let dep = depsMap.get(key);
  if (!dep) { depsMap.set(key, (dep = new Set())); }
  if (activeEffect && !dep.has(activeEffect)) {
    dep.add(activeEffect);
    activeEffect.deps.push(dep);  // 双向记录，用于 cleanup
  }
}

// 2. 派发更新 (trigger)
function trigger(target, key) {
  const depsMap = targetMap.get(target);
  if (!depsMap) return;
  const effects = new Set(depsMap.get(key) || []);
  effects.forEach(e => {
    if (e !== activeEffect) {   // 防止死循环
      if (e.scheduler) e.scheduler();  // computed / watch 走调度器
      else e.run();
    }
  });
}

// 3. effect（依赖的消费者）
function effect(fn, options = {}) {
  const _effect = new ReactiveEffect(fn, options.scheduler);
  _effect.run();
  return _effect;
}

1.2 ref vs reactive：什么时候用什么

refreactive
基础类型✅ 支持 string/number/boolean❌ 只支持对象
解构保持响应式❌ 解构后失去 (.value 是响应式的，但解构后变量不是)❌ 解构后失去（需 toRefs）
替换整个值✅ ref.value = newVal 触发更新❌ state = newObj 丢失响应式
template 解包✅ 自动解包（顶层 ref）—
底层实现class RefImpl { get value() { track(); return _value; } }new Proxy(target, handlers)
经验法则：单个值用 ref；一组关联状态用 reactive；API 返回值一律用 ref（方便解构和传递）。涉及解构用 toRefs。

1.3 computed 的实现——懒执行 + 缓存 + 依赖追踪

class ComputedRefImpl {
  constructor(getter) {
    this._dirty = true;           // 脏标记
    this._value = undefined;
    this.effect = new ReactiveEffect(getter, () => {
      // scheduler：依赖变化时不立即重算，只标脏
      if (!this._dirty) {
        this._dirty = true;
        trigger(this, 'value');  // 通知 computed 的消费者
      }
    });
  }
  get value() {
    track(this, 'value');
    if (this._dirty) {
      this._dirty = false;
      this._value = this.effect.run();  // 懒求值
    }
    return this._value;
  }
}

1.4 watch vs watchEffect vs computed 的选择

API触发时机有返回值适用场景
computed依赖变化时懒求值✅ 同步值派生状态（如 fullName = firstName + lastName）
watchEffect立即执行 + 依赖变化时自动执行❌无副作用的自动追踪（如 watchEffect(() => fetchData(id.value))
watch数据源变化时执行✅ 可拿到新旧值需对比新旧值、需条件触发（如路由变化）

2.2 Vue 3 Diff 算法核心优化（比 React 快的关键）

// Vue 3 patchChildren 五步法（针对无 key 或静态提升的场景）

// Step 1: 从头同步相同节点
while (i <= e1 && i <= e2) {
  const n1 = c1[i], n2 = c2[i];
  if (sameVnode(n1, n2)) { patch(n1, n2); i++; }
  else break;
}

// Step 2: 从尾同步相同节点
while (i <= e1 && i <= e2) {
  const n1 = c1[e1], n2 = c2[e2];
  if (sameVnode(n1, n2)) { patch(n1, n2); e1--; e2--; }
  else break;
}

// Step 3: 老的全遍历完了，新的还有 → 挂载剩余新节点
if (i > e1) { /* mount new nodes from i to e2 */ }

// Step 4: 新的全遍历完了，老的还有 → 卸载剩余老节点
if (i > e2) { /* unmount old nodes from i to e1 */ }

// Step 5: 中间乱序区域 → 最长递增子序列 (LIS)
// 这是核心优化：找到可复用的最长序列，只移动其他的
const newIdxMap = new Map();  // key → index (新 children)
for (let j = s2; j <= e2; j++) newIdxMap.set(c2[j].key, j);

// 遍历老的，标记哪些可复用、哪些要删
const newIdxToOldIdx = new Array(toBePatched);
for (let j = s1; j <= e1; j++) {
  const newIdx = newIdxMap.get(c1[j].key);
  if (newIdx !== undefined) {
    newIdxToOldIdx[newIdx - s2] = j;  // 记录对应关系
    patch(c1[j], c2[newIdx]);
  } else { unmount(c1[j]); }  // 老节点在新列表不存在
}

// 求 LIS，只移动不在 LIS 中的节点 → O(n log n) 的最少移动
const lis = getSequence(newIdxToOldIdx);
for (let j = lis.length - 1; j >= 0; j--) {
  if (lis.includes(j)) continue;    // 在 LIS 中 = 不需要移动
  const anchor = c2[nextIdx];          // 插入到下一个节点之前
  move(c2[j], anchor);
}
为什么用 LIS？旧的 children 的顺序 = newIdxToOldIdx 数组，找到其中的最长递增子序列 = 那些相对顺序已经对的节点，它们不需要移动。其余节点插入到正确位置。时间复杂度从 O(n²) 降到 O(n log n)。

2.3 Block Tree 与靶向更新（Vue 3 独有的大杀器）

// 传统 diff：全树递归，即使只有 1 个节点变了
// Vue 3 的 Block Tree：
// - 模板编译时，静态节点被提升（hoisted），不参与 diff
// - 动态节点收集到 dynamicChildren 数组
// - patch 时只遍历 dynamicChildren，跳过所有静态内容

// 编译前：
<div>
  <span>static text</span>              <!-- 不变 → hoist 外面 -->
  <p>{{ dynamic }}</p>               <!-- 动态 → 进 dynamicChildren -->
  <span class="static">more static</span> <!-- 不变 → hoist 外面 -->
</div>

// 编译后（简化）：
const _hoisted_1 = createVNode('span', null, 'static text');
const _hoisted_2 = createVNode('span', {class: 'static'}, 'more static');
render() {
  return (openBlock(), createBlock('div', null, [
    _hoisted_1,    // 静态：patchFlag = -1 (NEVER need patch)
    createVNode('p', null, ctx.dynamic, 1 /* TEXT */),
    _hoisted_2,
  ]));
}
// patchFlag 位掩码（编译优化核心）
const PatchFlags = {
  TEXT:          1,   // 只改文本
  CLASS:         2,   // 只改 class
  STYLE:         4,   // 只改 style
  PROPS:         8,   // 只改 props（不含 class/style）
  FULL_PROPS:    16,  // props 有动态 key → 需全量 diff
  NEED_HYDRATE:  32,  // 事件绑定
  STABLE_FRAG:   64,  // 子节点顺序稳定
  KEYED_FRAG:    128, // 子节点有 key
  UNKEYED_FRAG:  256, // 子节点无 key
  DYNAMIC_SLOTS: 512, // 动态插槽
};
// 通过 patchFlag，Vue 3 直接跳到具体要更新的属性类型，
// 跳过 vnode.type 的比较、全量 props diff 等冗余步骤。

3.2 transform：AST 优化的关键阶段

  • 静态提升（hoist）：不依赖响应式数据的 vnode 提到 render 函数外面，每次渲染复用同一对象
  • 预字符串化：连续纯静态节点（> 20 个）直接序列化成 innerHTML 字符串
  • 缓存事件处理函数：@click="handler" 编译成 _cache[0] || (_cache[0] = e => ctx.handler(e))
  • v-if / v-for 优化：由于 template 已知结构，Vue 可以推断静态部分

4.3 provide / inject 的实现

// 原理：组件实例上维护 provides 对象，向上查找原型链
// inject 沿着 parent 链往上找 provides 中的 key
function provide(key, value) {
  const instance = getCurrentInstance();
  let provides = instance.provides;
  const parentProvides = instance.parent?.provides;

  // 惰性原型继承：第一次 provide 时把 parent.provides 设为原型
  if (provides === parentProvides) {
    provides = instance.provides = Object.create(parentProvides);
  }
  provides[key] = value;
}

function inject(key, defaultValue) {
  const instance = getCurrentInstance();
  const provides = instance.parent.provides;
  return key in provides ? provides[key] : defaultValue;
}

五、Composition API 进阶模式

5.1 组合函数（composables）的设计原则

// ✅ 好的 composable：返回 ref，调用方可自由组合
export function useMouse() {
  const x = ref(0)
  const y = ref(0)
  const cleanup = () => {
    window.removeEventListener('mousemove', onMouseMove)
  }
  onMounted(() => window.addEventListener('mousemove', onMouseMove))
  onUnmounted(cleanup)

  return { x, y }  // 返回 ref → 调用方可 watch、computed、传给子组件
}

// ❌ 不好的 composable：返回普通值，失去响应式关联
export function useMouse() {
  let x = 0, y = 0;  // ❌ 纯数值，调用方 watch 不了
  return { x, y }
}

5.2 带副作用的 composable —— 自动清理

export function useEventListener(target, event, handler) {
  const cleanup = () => target.removeEventListener(event, handler)
  onMounted(() => target.addEventListener(event, handler))

  // effectScope：当调用方组件卸载时，自动停止此 scope 内的所有 effect
  const scope = effectScope()
  scope.run(() => {
    watch(source, handler)   // scope 内的 watch 会自动清理
    watchEffect(() => {})    // scope 内的 watchEffect 也会自动清理
  })
  onScopeDispose(() => scope.stop())  // 组件卸载时停止 scope
  onUnmounted(cleanup)
}
